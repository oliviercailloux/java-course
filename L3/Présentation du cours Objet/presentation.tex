\begin{frame}

  \begin{itemize}
		\item Savoir calculer, savoir coder
		\item Connaitre la plomberie sans faire de la plomberie
		\item Comprendre et critiquer
	\end{itemize}
\end{frame}

\subsection{Évaluation}
\begin{frame}
	\frametitle{Évaluation}
	100\% CC
	\begin{itemize}
		\item ≈ 4 tests en séance (annoncés)
		\item Devoirs maison : contributions individuelles à un projet commun
		% \item (Possible selon avancement)  QCMs
		\item Aggrégation {\tiny (\og{}\href{https://en.wikipedia.org/wiki/Ordered_weighted_averaging}{owa}\fg)} des notes reçues au long de l’année
		\item Pondérations tests à déterminer ; exemple : 1 pour git et au mieux par individu parmi $\set{1, 2, 3}$
    \item Pondérations projet VS tests : au pire par individu parmi $\set{1/5, 4/5}$
    % \begin{description}[Projet VS tests]
    %   \item[tests 1 à 6] au mieux par individu parmi $\set{1/10, 1/10,\allowbreak 1,\allowbreak 1,\allowbreak 1,\allowbreak 1}$ {\tiny (plafond possible pour tests simples)}, 
    %   \item[dernier test] $2$
      % \item[projet VS tests] au pire par individu parmi $\set{1/5, 4/5}$
    % \end{description}
  \end{itemize}
\end{frame}

\begin{frame}
	\frametitle{Évaluation des tests}
  Avec ressources pré-existantes, sans IA générative
	\begin{itemize}
		\item Code doit compiler
		\item Livraison via git
		\item Respect précis des instructions
		\item Évaluation généralement automatique
		\item Points pour chaque aspect fonctionnel
		\item Rattrapage possible avec pénalité si solution presque fonctionnelle
	\end{itemize}
\end{frame}

\begin{frame}
	\frametitle{Projet}
  IA génératives autorisées
	\begin{itemize}
		\item Projet commun : jeu de \href{https://world-geography-games.com/en/countries_world.html}{localisation géographique} de pays
		\item Architecture client-serveur REST
		\item Outils modernes : Quarkus, Docker…
		\item À apprendre en autonomie !
		\item Bloquer deux dates de livraison, espacées d’au moins trois semaines
		\item Deux volontaires intégration (considérez le \href{https://doi.org/10.1093/ej/uez050}{biais de genre})
		\item Livraison : indiquer sur devoir Moodle le lien \href{https://docs.github.com/en/pull-requests/committing-changes-to-your-project/viewing-and-comparing-commits/comparing-commits\#comparing-commits}{diff}
	\end{itemize}
\end{frame}
